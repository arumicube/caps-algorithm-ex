#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#define SIZE 7
#define MAX 100

int main() {
    int n;
    scanf_s("%d", &n);
    int people[MAX][SIZE];

    for (int i = 0; i < n; i++) {
        int p;
        scanf_s("%d", &p);
        for (int j = SIZE - 1; j >= 0; j--) {
            people[i][j] = p % 2;
            p /= 10;
        }
    }

    int CommonDay[SIZE];
    for (int i = 0; i < SIZE; i++) {
        CommonDay[i] = 1;
        for (int j = 0; j < n; j++) {
            CommonDay[i] &= people[j][i];
        }
    }

    for (int i = 0; i < SIZE; i++) {
        if (CommonDay[i]) {
            switch (i) {
            case 0:
                printf("월요일에 가능\n");
                break;
            case 1:
                printf("화요일에 가능\n");
                break;
            case 2:
                printf("수요일에 가능\n");
                break;
            case 3:
                printf("목요일에 가능\n");
                break;
            case 4:
                printf("금요일에 가능\n");
                break;
            case 5:
                printf("토요일에 가능\n");
                break;
            case 6:
                printf("일요일에 가능\n");
                break;
            default:
                printf("다음에 만나용~");
                break;
            }
        }
        else {
            if (i == SIZE-1) {
                printf("다음에 만나용~");
                break;
            }
        }
    }

    return 0;
}
